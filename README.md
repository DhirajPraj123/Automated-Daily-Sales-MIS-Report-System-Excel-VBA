'====================================================================
'  DAILY SALES MIS — DYNAMIC AUTOMATION MODULE
'  NO USERFORM REQUIRED — Works directly with SALES_DATA sheet
'====================================================================

Option Explicit

Const SRC   As String = "SALES_DATA"
Const DASH  As String = "MIS_DASHBOARD"
Const RRPT  As String = "REGION_REPORT"
Const PRPT  As String = "PRODUCT_REPORT"
Const DROW  As Long = 4

Sub MISRuns()

Application.ScreenUpdating = False
Application.Calculation = xlCalculationManual
Application.EnableEvents = False


Call ValidateData
Call RefreshDashboard
Call GenerateRegionReport
Call GenerateProductReport
Call HighlightTopPerformer
Call AddSummaryTimestamp


Application.Calculation = xlCalculationAutomatic
Application.ScreenUpdating = True
Application.EnableEvents = True

MsgBox "MIS Report Generated Successfully!" & vbNewLine & _
"Dashboard, Region & Product reports updated.", _
vbInformation, "MIS Automation Complete"
End Sub

'  ValidateData — checks SALES_DATA for blank/invalid cells


Sub ValidateData()
    Dim ws As Worksheet: Dim lastRow As Long
    Dim i As Long: Dim errors As String: Dim errCount As Long
    Set ws = ThisWorkbook.Sheets(SRC)
    lastRow = ws.Cells(ws.Rows.Count, 2).End(xlUp).Row

    For i = DROW To lastRow
        If Trim(ws.Cells(i, 2).Value) = "" Then
            errors = errors & "Row " & i & ": Date missing" & vbNewLine
            errCount = errCount + 1
        End If
       If Trim(ws.Cells(i, 3).Value) = "" Then
           errors = errors & "Row " & i & ": Salesperson missing" & vbNewLine
           errCount = errCount + 1
       End If
       If Not IsNumeric(ws.Cells(i, 6).Value) Then
           errors = errors & "Row " & i & ": Qty invalid" & vbNewLine
           errCount = errCount + 1
       End If
       If Not IsNumeric(ws.Cells(i, 7).Value) Then
           errors = errors & "Row " & i & ": Price invalid" & vbNewLine
           errCount = errCount + 1
       End If
   Next i

   If errCount > 0 Then
       MsgBox "DATA ERRORS FOUND: " & errCount & " issues:" & _
              vbNewLine & errors, vbExclamation, "Validation Failed"
       End  '? stops entire macro. Fix data first, then run again
   End If
End Sub

   

'--------------------------------------------------------------------
'  RefreshDashboard — forces recalculation of all dashboard formulas
'--------------------------------------------------------------------

Sub RefreshDashboard()

    ThisWorkbook.Sheets(DASH).Calculate
    ThisWorkbook.Sheets(DASH).Cells(2, 2).Value = _
            "Auto-generated by VBA  |  Updated: " & Format(Now, "DD-MMM-YYYY HH:MM")
End Sub
Sub GenerateRegionReport()
    Dim wsSrc As Worksheet, wsRpt As Worksheet
    Dim lastRow As Long, i As Long, rRow As Long
    Set wsSrc = ThisWorkbook.Sheets(SRC)
    Set wsRpt = ThisWorkbook.Sheets(RRPT)
    lastRow = wsSrc.Cells(wsSrc.Rows.Count, 2).End(xlUp).Row

    wsRpt.Rows("4:1000").ClearContents
    wsRpt.Rows("4:1000").Interior.ColorIndex = xlNone
    rRow = 4

   Dim sortedData() As Variant
   ReDim sortedData(1 To lastRow - DROW + 1, 1 To 8)

   For i = DROW To lastRow
       wsRpt.Cells(rRow, 2).Value = wsSrc.Cells(i, 4).Value
       wsRpt.Cells(rRow, 3).Value = wsSrc.Cells(i, 3).Value
       wsRpt.Cells(rRow, 4).Value = wsSrc.Cells(i, 5).Value
       wsRpt.Cells(rRow, 5).Value = wsSrc.Cells(i, 6).Value
       wsRpt.Cells(rRow, 6).Value = wsSrc.Cells(i, 8).Value
       wsRpt.Cells(rRow, 7).Value = wsSrc.Cells(i, 9).Value

       Dim sales As Double, tgt As Double
       sales = wsSrc.Cells(i, 8).Value
       tgt = wsSrc.Cells(i, 9).Value

       If sales >= tgt Then
           wsRpt.Cells(rRow, 8).Value = ChrW(10004) & " HIT"
           wsRpt.Cells(rRow, 8).Font.Color = RGB(30, 107, 60)
           wsRpt.Cells(rRow, 8).Interior.Color = RGB(212, 237, 218)
       Else
           wsRpt.Cells(rRow, 8).Value = ChrW(10008) & " MISS"
           wsRpt.Cells(rRow, 8).Font.Color = RGB(139, 26, 26)
           wsRpt.Cells(rRow, 8).Interior.Color = RGB(253, 222, 222)
       End If

       wsRpt.Cells(rRow, 6).NumberFormat = "#,##0"
       wsRpt.Cells(rRow, 7).NumberFormat = "#,##0"

       If (rRow Mod 2) = 0 Then
           wsRpt.Range(wsRpt.Cells(rRow, 2), wsRpt.Cells(rRow, 7)).Interior.Color = RGB(244, 244, 244)
       End If

       rRow = rRow + 1
   Next i
End Sub


'--------------------------------------------------------------------
'  GenerateProductReport — builds PRODUCT_REPORT summary
'--------------------------------------------------------------------
Sub GenerateProductReport()
    Dim wsSrc As Worksheet, wsRpt As Worksheet
    Dim lastRow As Long, i As Long
    Set wsSrc = ThisWorkbook.Sheets(SRC)
    Set wsRpt = ThisWorkbook.Sheets(PRPT)
    lastRow = wsSrc.Cells(wsSrc.Rows.Count, 2).End(xlUp).Row

    Dim products(2) As String
    products(0) = "Laptop Pro X1"
    products(1) = "Office Chair Pro"
    products(2) = "Wireless Mouse"

   Dim totalRev As Double: totalRev = 0
   Dim p As Integer, j As Long
   Dim prodQty(2) As Long, prodSales(2) As Double, prodOrders(2) As Long

   For j = DROW To lastRow
       For p = 0 To 2
           If wsSrc.Cells(j, 5).Value = products(p) Then
               prodQty(p) = prodQty(p) + wsSrc.Cells(j, 6).Value
               prodSales(p) = prodSales(p) + wsSrc.Cells(j, 8).Value
               prodOrders(p) = prodOrders(p) + 1
               totalRev = totalRev + wsSrc.Cells(j, 8).Value
           End If
       Next p
   Next j

   wsRpt.Rows("4:6").ClearContents

   For p = 0 To 2
       Dim r As Long: r = 4 + p
       wsRpt.Cells(r, 2).Value = products(p)
       wsRpt.Cells(r, 3).Value = prodQty(p)
       wsRpt.Cells(r, 4).Value = prodSales(p)
       If prodQty(p) > 0 Then
           wsRpt.Cells(r, 5).Value = prodSales(p) / prodOrders(p)
       End If
       wsRpt.Cells(r, 6).Value = prodOrders(p)
       If totalRev > 0 Then
           wsRpt.Cells(r, 7).Value = prodSales(p) / totalRev
           wsRpt.Cells(r, 7).NumberFormat = "0.0%"
       End If
   Next p
End Sub

'--------------------------------------------------------------------
'  HighlightTopPerformer — finds best salesperson and highlights row
'--------------------------------------------------------------------
Sub HighlightTopPerformer()
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(DASH)
    Dim maxSales As Double: Dim maxRow As Long: Dim r As Long
    maxSales = 0

    For r = 10 To 14
        ws.Rows(r).Interior.ColorIndex = xlNone
    Next r

    For r = 10 To 14
        If ws.Cells(r, 8).Value > maxSales Then
            maxSales = ws.Cells(r, 8).Value
           maxRow = r
       End If
   Next r

   If maxRow > 0 Then
       ws.Range(ws.Cells(maxRow, 7), ws.Cells(maxRow, 10)).Interior.Color = RGB(255, 243, 185)
       ws.Range(ws.Cells(maxRow, 7), ws.Cells(maxRow, 10)).Font.Bold = True
   End If
End Sub

'--------------------------------------------------------------------
'  AddSummaryTimestamp — stamps run time on dashboard
'--------------------------------------------------------------------
Sub AddSummaryTimestamp()
    Dim ws As Worksheet: Set ws = ThisWorkbook.Sheets(DASH)
    ws.Cells(2, 2).Value = "Auto-generated by VBA  |  Period: May 2026  |  Updated: " & _
        Format(Now, "DD-MMM-YYYY HH:MM AM/PM")
End Sub

'--------------------------------------------------------------------
'  ClearReports — wipes Region & Product report data only
'--------------------------------------------------------------------

Sub ClearReports()
    Dim ans As Integer
    ans = MsgBox("Clear all generated reports? Source data will NOT be deleted.", _
         vbQuestion + vbYesNo, "Confirm Clear")
    If ans = vbNo Then Exit Sub

    ThisWorkbook.Sheets(RRPT).Rows("4:1000").ClearContents
    ThisWorkbook.Sheets(RRPT).Rows("4:1000").Interior.ColorIndex = xlNone
    ThisWorkbook.Sheets(PRPT).Rows("4:6").ClearContents
    MsgBox "Reports cleared. Source data is intact.", vbInformation, "Cleared"
End Sub


End Function
